\section{Problem 1: The Truncated Parabola in a Harmonic Oscillator}
\label{sec:problem1}

\subsection*{Problem Statement}
Consider a particle of mass $m$ subject to a one-dimensional quantum harmonic oscillator potential:
\begin{equation}
    V(x) = \frac{1}{2}m\omega^2x^2
\end{equation}

At time $t=0$, the particle is prepared in the initial state:
\begin{equation}
    \psi(x,0) = 
    \begin{cases} 
        N(A - Bx^2) & \text{for } x^2 \le \frac{A}{B} \\ 
        0 & \text{for } x^2 > \frac{A}{B} 
    \end{cases}
\end{equation}
where $A$ and $B$ are positive real constants, and $N$ is the normalization constant.

\subsection*{Analytical Tasks}
\begin{enumerate}[label=\textbf{\arabic*.}]
    \item \textbf{Normalization Constant:} Determine the normalization constant $N$ in terms of $A$ and $B$ by enforcing the normalization condition:
    \begin{equation}
        \int_{-\infty}^{\infty} |\psi(x,0)|^2 \dd{x} = 1
    \end{equation}
    
    \item \textbf{Eigenstate Expansion:} Calculate the probability $P_n = |c_n|^2$ of finding the particle in the energy eigenstate $\ket{\phi_n}$ with energy eigenvalue $E_n = \left(n + \frac{1}{2}\right)\hbar\omega$. The expansion coefficients are given by the projection:
    \begin{equation}
        c_n = \braket{\phi_n}{\psi(x,0)} = \int_{-\sqrt{A/B}}^{\sqrt{A/B}} \phi_n^*(x) \psi(x,0) \dd{x}
    \end{equation}
    where $\phi_n(x)$ are the Hermite-Gaussian eigenfunctions of the quantum harmonic oscillator.

    \item \textbf{Time Evolution:} Write down the formal expression for the time-evolved wave function $\psi(x,t)$ for $t > 0$:
    \begin{equation}
        \psi(x,t) = \sum_{n=0}^{\infty} c_n \phi_n(x) e^{-i E_n t / \hbar}
    \end{equation}
\end{enumerate}

\section{Problem 2: The Shifted and Squeezed Gaussian}
\label{sec:problem2}

\subsection*{Problem Statement}
A particle of mass $m$ in a harmonic oscillator potential $V(x) = \frac{1}{2}m\omega^2x^2$ is initially prepared in a displaced (shifted) and squeezed Gaussian wave packet:
\begin{equation}
    \psi(x,0) = N e^{-\frac{(x-x_0)^2}{a^2}}
\end{equation}
where:
\begin{itemize}
    \item $x_0$ is the initial spatial displacement (shift) of the wave packet center.
    \item $a$ is the initial width of the wave packet, such that $a \neq \sqrt{\frac{\hbar}{m\omega}}$ (meaning the wave packet width differs from the ground-state characteristic length scale $x_c = \sqrt{\frac{\hbar}{m\omega}}$, resulting in a squeezed state).
    \item $N$ is the normalization constant.
\end{itemize}

\subsection*{Analytical Tasks}
\begin{enumerate}[label=\textbf{\arabic*.}]
    \item \textbf{Time Evolution:} Determine the full time-evolved wave packet $\psi(x,t)$ under the harmonic oscillator Hamiltonian. Show that the wave packet remains Gaussian in form:
    \begin{equation}
        \psi(x,t) = N(t) e^{-A(t)(x - \langle x \rangle(t))^2 + \frac{i}{\hbar} \langle p \rangle(t) x + i \phi(t)}
    \end{equation}
    where $A(t)$ is a complex parameter related to the time-dependent width and phase.

    \item \textbf{Expectation Values:} Calculate the expectation values of position $\langle x \rangle(t)$ and momentum $\langle p \rangle(t)$ as functions of time. Verify that they satisfy Ehrenfest's theorem and display classical oscillatory behavior:
    \begin{align}
        \langle x \rangle(t) &= x_0 \cos(\omega t) \\
        \langle p \rangle(t) &= -m\omega x_0 \sin(\omega t)
    \end{align}

    \item \textbf{Uncertainty Relations and Breathing Mode:} 
    Determine the position uncertainty $\Delta x(t) = \sqrt{\langle x^2 \rangle - \langle x \rangle^2}$ and momentum uncertainty $\Delta p(t)$. Show that the wave packet width oscillates (or "breathes") at a frequency of $2\omega$.
\end{enumerate}

\section{Problem 3: Tunneling in the Quartic Double Well}
\label{sec:problem3}

\subsection*{Problem Statement}
A particle of mass $m$ moves in a one-dimensional quartic double-well potential described by:
\begin{equation}
    V(x) = -\frac{1}{2}\alpha x^2 + \beta x^4
\end{equation}
where $\alpha$ and $\beta$ are positive constants. This potential features two symmetric minima at $x_{\text{min}} = \pm \sqrt{\frac{\alpha}{4\beta}}$ separated by a potential barrier of height $V_0 = \frac{\alpha^2}{16\beta}$ at $x = 0$.

\subsection*{Physical and Computational Tasks}
\begin{enumerate}[label=\textbf{\arabic*.}]
    \item \textbf{Eigenstate Calculation:} Find the lowest energy eigenstates by solving the time-independent Schrödinger equation:
    \begin{equation}
        \left[ -\frac{\hbar^2}{2m} \frac{\dd^2}{\dd{x}^2} + V(x) \right] \psi_n(x) = E_n \psi_n(x)
    \end{equation}
    numerically on a discretized spatial grid of size $N$ over the domain $x \in [-L, L]$ using finite difference approximations.

    \item \textbf{Energy Eigenvalues and Degeneracy Splitting:} 
    Determine the ground state energy $E_0$ (symmetric state, even parity) and the first excited state energy $E_1$ (antisymmetric state, odd parity). Calculate the energy splitting:
    \begin{equation}
        \Delta E = E_1 - E_0
    \end{equation}
    Explain how quantum tunneling lifts the classical degeneracy of the two isolated wells.

    \item \textbf{Tunneling Time:} 
    Construct a state initially localized in the right well:
    \begin{equation}
        \psi_{R}(x,0) = \frac{1}{\sqrt{2}} \left( \psi_0(x) + \psi_1(x) \right)
    \end{equation}
    Show that the wave function evolves in time as:
    \begin{equation}
        \psi(x,t) = \frac{1}{\sqrt{2}} \left( \psi_0(x)e^{-iE_0t/\hbar} + \psi_1(x)e^{-iE_1t/\hbar} \right)
    \end{equation}
    Calculate the tunneling time $\tau$ required for the particle to tunnel completely from the right well to the left well, where:
    \begin{equation}
        \tau = \frac{\pi\hbar}{\Delta E}
    \end{equation}

    \item \textbf{Grid Convergence Study:}
    Conduct a numerical grid convergence test by solving the eigenvalue problem for varying spatial discretization grid points
    \[
        N \in \{200,\; 400,\; 800,\; 1600,\; 3200\}
    \]
    to verify the stability and accuracy of the computed eigenvalues $E_0$ and energy splitting $\Delta E$.
\end{enumerate}

\section{Problem 4: Numerical Investigation of Quantum Tunneling in a Quartic Double-Well Potential using the \textbf{Eigenstate Expansion}}
\textit{Notebook: \texttt{04 Quantum Tunneling in Double Well Potential.ipynb}}
\label{sec:problem4}


The one-dimensional time-independent Schrödinger equation describes the quantum behavior of a particle of mass $m$ moving under the influence of a potential $V(x)$,

\begin{equation}
-\frac{\hbar^2}{2m}\frac{d^2\psi(x)}{dx^2}+V(x)\psi(x)=E\psi(x),
\end{equation}

where $\psi(x)$ is the wavefunction and $E$ is the corresponding energy eigenvalue. Analytical solutions to this equation exist only for a limited number of idealized potentials. For more realistic systems, numerical techniques become indispensable.

One important quantum system is the quartic double-well potential,

\begin{equation}
V(x)=-\frac{1}{2}\alpha x^2+\beta x^4,
\end{equation}

which exhibits two stable minima separated by a finite potential barrier. Although classically a particle localized in one well cannot cross the barrier if its energy is below the barrier height, quantum mechanics predicts the phenomenon of tunneling, whereby the particle has a finite probability of appearing in the opposite well. This phenomenon gives rise to characteristic energy-level splitting and time-dependent oscillatory dynamics.

The primary objective of this work is to develop and validate a finite-difference numerical solver for the one-dimensional Schrödinger equation and subsequently employ it to investigate the stationary and dynamical properties of a particle confined in a quartic double-well potential.

The study is divided into three stages.

\subsection{Stage I: Validation of the Numerical Solver}

The numerical solver will first be validated using the harmonic oscillator potential,

\begin{equation}
V(x)=\frac{1}{2}m\omega^2x^2,
\end{equation}

whose exact analytical solutions are well established. The validation consists of

\begin{enumerate}
    \item Computing the first twenty energy eigenvalues.
    \item Computing the corresponding twenty eigenfunctions.
    \item Constructing an error table by comparing numerical and analytical eigenvalues.
    \item Plotting numerical and analytical energy spectra for comparison.
    \item Comparing numerical and analytical eigenfunctions graphically.
\end{enumerate}

Successful agreement with the analytical solution verifies the correctness, accuracy, and convergence of the numerical implementation.

\subsection{Stage II: Eigenstate Analysis of the Quartic Double-Well Potential}

Following validation, the numerical solver will be applied to the quartic double-well potential,

\begin{equation}
V(x)=-\frac{1}{2}\alpha x^2+\beta x^4.
\end{equation}

The objectives of this stage are

\begin{enumerate}
    \item Compute the first twenty eigenvalues.
    \item Compute the corresponding twenty eigenfunctions.
    \item Perform a grid-convergence study to verify numerical stability.
    \item Visualize the potential together with the bound-state wavefunctions.
    \item Analyze the energy spectrum, with particular emphasis on tunneling-induced energy splitting.
\end{enumerate}

\subsection{Stage III: Gaussian Wave Packet Dynamics}

To investigate quantum tunneling dynamically, an initial Gaussian wave packet will be constructed near the left minimum of the double-well potential,

\begin{equation}
\Psi(x,0)
=
A\,
\exp\!\left[
-\frac{(x-x_0)^2}{4\sigma^2}
\right]
e^{ik_0x},
\end{equation}

where $A$ is the normalization constant, $x_0=-\sqrt{\alpha/4\beta}$ is the left potential minimum, $\sigma=0.5$ is the packet width, and $k_0=0$ (particle initially at rest).

The initial wave packet will be expanded in terms of the stationary eigenstates,

\begin{equation}
\Psi(x,0)=\sum_{n=0}^{N-1} c_n\phi_n(x),
\end{equation}

where the expansion coefficients are

\begin{equation}
c_n
=
\int_{-\infty}^{\infty}
\phi_n^*(x)\Psi(x,0)\,dx.
\end{equation}

The subsequent time evolution is obtained from

\begin{equation}
\Psi(x,t)
=
\sum_{n=0}^{N-1}
c_n
\phi_n(x)
e^{-iE_nt/\hbar}.
\end{equation}

The dynamical investigation consists of

\begin{enumerate}
    \item Locating the left minimum: $x_0 = -\sqrt{\alpha/4\beta}$.
    \item Constructing and normalizing the Gaussian wave packet.
    \item Expanding in the stationary eigenstate basis and computing $c_n$.
    \item Performing a completeness check: $\displaystyle\sum_{n=0}^{19}|c_n|^2 = 0.9992$, confirming that 20 states capture $99.9\%$ of the packet.
    \item Plotting $|c_n|^2$, which shows that $|c_0|^2\approx|c_1|^2\approx 0.49$, establishing the two-state dominance of the tunneling dynamics.
    \item Computing the time evolution $\Psi(x,t)$ and verifying reconstruction at $t=0$: reconstruction error $= 2.75\times10^{-2}$, consistent with the $0.08\%$ basis leakage.
    \item Determining the tunneling period from $\Delta E = E_1 - E_0 \approx 1.49\times10^{-5}$:
    \begin{equation}
        T = \frac{2\pi\hbar}{\Delta E} \approx 420{,}776.
    \end{equation}
    \item Animating $|\Psi(x,t)|^2$ over one full tunneling period.
    \item Plotting $P_{\text{left}}(t)$ and $P_{\text{right}}(t)$ over two full cycles, confirming sinusoidal anti-correlated oscillation with the computed period $T$.
\end{enumerate}

\subsection{Stage IV: Deep-Well Simplification}

To probe the professor's remark that ``things should simplify for a deep well,'' the analysis is repeated for $\alpha \in \{10,\,20,\,50\}$ with $\beta=0.5$ fixed. As $\alpha$ increases:

\begin{enumerate}
    \item The barrier height $V_0 = \alpha^2/16\beta$ grows, making the well deeper and the barrier wider.
    \item The energy splitting $\Delta E$ shrinks exponentially (WKB prediction: $\Delta E\propto e^{-\gamma}$ where $\gamma=\int\sqrt{2m(V-E)}\,dx/\hbar$).
    \item The basis weight concentrates: $|c_0|^2+|c_1|^2\to 1$, so the full twenty-state expansion
    \begin{equation}
        \Psi(x,t)=\sum_{n=0}^{19}c_n\psi_n(x)\,e^{-iE_nt/\hbar}
    \end{equation}
    collapses into the exact two-state problem
    \begin{equation}
        \Psi(x,t)\approx c_0\psi_0(x)\,e^{-iE_0t/\hbar}+c_1\psi_1(x)\,e^{-iE_1t/\hbar}.
    \end{equation}
\end{enumerate}

Physically, tunneling in a deep well is governed by a single number $\Delta E$: the dynamics are slower, cleaner, and entirely describable by the interference of the symmetric and antisymmetric ground states.

The successful completion of all four stages yields a comprehensive numerical framework. The key findings are summarised below.

\begin{itemize}
    \item \textbf{Validation (Stage~I):} Numerical eigenvalues agree with the exact harmonic-oscillator spectrum $E_n=(n+\tfrac{1}{2})\hbar\omega$ to within $\sim10^{-4}$ for the lowest twenty states, with error growing with $n$ as expected from finite-difference discretization.
    \item \textbf{Grid convergence (Stage~II):} Eigenvalues of the double well stabilize by $N=1600$ grid points, establishing grid independence. The ground-state doublet splitting $\Delta E\approx1.49\times10^{-5}$ is the numerical signature of quantum tunneling.
    \item \textbf{Wave-packet dynamics (Stage~III):} The Gaussian expands almost entirely into the two lowest eigenstates. The resulting probability density oscillates between the wells with period $T\approx420{,}776$ in natural units, confirmed by both the animation and the left/right probability plot.
    \item \textbf{Deep-well limit (Stage~IV):} Increasing $\alpha$ drives $\Delta E\to 0$ exponentially and $|c_0|^2+|c_1|^2\to1$, reducing the dynamics to a pure two-state system and confirming the theoretical prediction.
\end{itemize}

\noindent In summary, quantum tunneling is the constructive and destructive interference of two nearly degenerate energy eigenstates; it can be fully understood, computed, and visualised using finite-difference diagonalization and eigenstate expansion on a spatial grid.
